-- 创建数据库
CREATE DATABASE IF NOT EXISTS pixel_art_db;
USE pixel_art_db;

-- 用户表
CREATE TABLE users (
    user_id INT AUTO_INCREMENT PRIMARY KEY,
    account VARCHAR(50) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL,
    username VARCHAR(50) UNIQUE NOT NULL,
    currency_amount INT DEFAULT 0
);

-- 图画表
CREATE TABLE drawings (
    drawing_id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) NOT NULL,
    FOREIGN KEY (username) REFERENCES users(username) ON DELETE CASCADE
);

-- 像素表
CREATE TABLE pixels (
    pixel_id INT AUTO_INCREMENT PRIMARY KEY,
    drawing_id INT,
    user_id INT NOT NULL,
    coordinates VARCHAR(20) NOT NULL,
    color VARCHAR(7) NOT NULL,
    FOREIGN KEY (drawing_id) REFERENCES drawings(drawing_id) ON DELETE SET NULL,
    FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE CASCADE
);
